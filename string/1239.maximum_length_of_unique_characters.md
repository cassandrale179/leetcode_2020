import collections
class Solution:
    def maxLength(self, arr: List[str]) -> int:
        keys = []
        for word in arr:
            if len(set(word)) == len(word):
                keys.append(word)
        answer = 0 
        for a in arr:
            for b in keys[:]:
                new_word = a + b
                if len(set(new_word)) == len(new_word):
                    keys.append(new_word)
        if len(keys) == 0:
            return 0
        return max(len(a) for a in keys)
